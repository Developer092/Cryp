#include <stdio.h>

int main() {
    char *str = "Hello world";

    printf("AND with 127: ");
    for (int i = 0; str[i] != '\0'; i++) {
        char a = str[i] & 127;
        printf("%c", a);
    }

    printf("\nXOR with 127: ");
    for (int i = 0; str[i] != '\0'; i++) {
        char x = str[i] ^ 127;
        printf("%c", x);
    }

    return 0;
}
