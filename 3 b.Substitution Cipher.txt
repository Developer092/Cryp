

import java.util.Scanner;

public class SubstitutionCipher {

    // Encrypt plaintext using substitution key
    public static String encrypt(String text, String key) {
        text = text.toUpperCase();
        key = key.toUpperCase();

        StringBuilder result = new StringBuilder();
        
        for (char c : text.toCharArray()) {
            if (c >= 'A' && c <= 'Z') {
                int index = c - 'A';
                result.append(key.charAt(index));
            } else {
                result.append(c);  // Keep spaces & symbols unchanged
            }
        }
        
        return result.toString();
    }

    // Decrypt ciphertext using substitution key
    public static String decrypt(String text, String key) {
        text = text.toUpperCase();
        key = key.toUpperCase();

        StringBuilder result = new StringBuilder();

        // Reverse map for quick lookup
        char[] reverseMap = new char[26];
        for (int i = 0; i < 26; i++) {
            reverseMap[key.charAt(i) - 'A'] = (char) ('A' + i);
        }

        for (char c : text.toCharArray()) {
            if (c >= 'A' && c <= 'Z') {
                result.append(reverseMap[c - 'A']);
            } else {
                result.append(c);
            }
        }

        return result.toString();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Example key: QWERTYUIOPASDFGHJKLZXCVBNM
        System.out.print("Enter substitution key (26 letters A-Z): ");
        String key = sc.nextLine();

        System.out.print("Enter text: ");
        String text = sc.nextLine();

        String encrypted = encrypt(text, key);
        String decrypted = decrypt(encrypted, key);

        System.out.println("\nEncrypted Text: " + encrypted);
        System.out.println("Decrypted Text: " + decrypted);

        sc.close();
    }
}
